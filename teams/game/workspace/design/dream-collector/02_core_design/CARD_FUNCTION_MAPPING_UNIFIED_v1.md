## 📋 **코드 구현 가이드**

### **GDScript: 기능 ID 검색**

```gdscript
class_name CardFunctionMapping

# 기능 ID → 카드 ID 배열
var function_to_cards: Dictionary = {
    "ATK-SGL": ["ATK_001", "ATK_006", "ATK_008", "ATK_010"],
    "ATK-MLT": ["ATK_002", "ATK_003", "ATK_007"],
    "ATK-RCK": ["ATK_004"],
    "ATK-CMP": ["ATK_005", "ATK_009"],
    
    "SKL-GRD": ["DEF_001", "DEF_002", "DEF_003", "DEF_004", "DEF_005", "DEF_006", "DEF_007", "DEF_008"],
    "SKL-PAR": ["PAR_001", "PAR_002", "PAR_003", "PAR_004", "PAR_005"],
    "SKL-DOD": ["DOD_001", "DOD_002", "DOD_003", "DOD_004", "DOD_005"],
    
    "PWR-DRW": ["SKL_001", "SKL_002"],
    "PWR-ATK": ["POW_001", "POW_002"],
    "PWR-DEF": ["POW_003", "POW_004"],
    "PWR-SUS": ["POW_005", "POW_006"],
    
    "CRS-STA": ["CUR_001", "CUR_002"],
    "CRS-SPD": ["CUR_003"],
    "CRS-PEN": ["CUR_004", "CUR_005"],
    "CRS-RSK": ["CUR_006"],
}

# 카드 ID → 기능 ID 역매핑
var card_to_function: Dictionary = {}

func _ready():
    # 역매핑 생성
    for func_id in function_to_cards:
        for card_id in function_to_cards[func_id]:
            card_to_function[card_id] = func_id

# 사용 예시
func get_function_id(card_id: String) -> String:
    return card_to_function.get(card_id, "UNKNOWN")

func get_cards_for_function(func_id: String) -> Array[String]:
    return function_to_cards.get(func_id, [])

# SKL-GRD 모든 카드 조회
var guard_cards = get_cards_for_function("SKL-GRD")  # 8장
```
