# Example 1: Data Generation
def generate_cards(count: int = 85) -> List[Dict]:
    """Generate card data"""
    cards = []
    for i in range(1, count + 1):
        card = {
            "id": i,
            "name_ko": f"카드_{i:03d}",
            "name_en": f"Card_{i:03d}",
            "cost": (i % 5) + 1,
            "type": ["Attack", "Defend", "Magic"][i % 3],
            "description": f"Card {i} description"
        }
        cards.append(card)
    return cards

# Example 2: File Processing
def save_to_json(data: Dict, file_path: str) -> bool:
    """Save data to JSON file with error handling"""
    try:
        path = Path(file_path)
        path.parent.mkdir(parents=True, exist_ok=True)
        with open(path, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        return True
    except IOError as e:
        print(f"Error writing file: {e}")
        return False

# Example 1: Build Test
measure_build_time() {
    local start=$(date +%s%N)
    
    # Build command
    godot-build.sh
    
    local end=$(date +%s%N)
    local duration=$(( ($end - $start) / 1000000 ))
    echo "Build time: ${duration}ms"
}

# Example 2: File Operations
process_files() {
    local input_dir="$1"
    local output_dir="$2"
    
    for file in "$input_dir"/*.json; do
        [ -f "$file" ] || continue
        log_info "Processing $(basename "$file")"
        # Process file
        cp "$file" "$output_dir/"
    done
}

# Common Issues & Solutions

Q: Module not found (Python)?
A: Install dependencies: pip install -r requirements.txt
   Or install specific: pip install requests

Q: Permission denied (Bash)?
A: chmod +x script.sh
   Or run: bash script.sh (don't use ./script.sh)

Q: Path not found?
A: Use absolute path: ~/Projects/geekbrox/...
   Use expanduser: os.path.expanduser("~/...")

Q: Script hangs?
A: Add timeouts or check for infinite loops
   Use set -u to catch undefined variables

Q: How to test?
A: Python: python3 script.py
   Bash: bash script.sh /path/to/input
   Check: ls -la output/
